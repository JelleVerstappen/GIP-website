Code aansturen met joystick



Dit is de code voor het aansturen van de auto via joystick. De X- en Y-co√∂rdinaten van de joystick worden bijgehouden, afhankelijk daar van beweegt de auto. 


      

                    int JoyStick_X = A0; // X-axis signal
                    int JoyStick_Y = A1; // Y-axis signal
                    int Button = 3; // Button
                     
                    void setup ()
                    {
                      pinMode (JoyStick_X, INPUT);
                      pinMode (JoyStick_Y, INPUT);
                      pinMode (Button, INPUT);
                       
                      // Since the button pulls the signal to ground when pressed,
                      // here we switch on the pullup resistor
                      digitalWrite(Button, HIGH);  
                       
                      Serial.begin (9600); // Serial output with 9600 bps
                    }
                     
                    // The program reads the current values of the input pins
                    // and outputs them to the serial output
                    void loop ()
                    {
                      float x, y;
                      int button;
                       
                      //Actual values are read, converted to the voltage value....
                      x = analogRead (JoyStick_X) * (5.0 / 1023.0); 
                      y = analogRead (JoyStick_Y) * (5.0 / 1023.0);
                      button = digitalRead (Button);
                       
                      //... and output at this position
                      Serial.print ("X axis:"); Serial.print (x, 4); Serial.print ("V, ");
                      Serial.print ("Y axis:"); Serial.print (y, 4); Serial.print ("V, ");
                      Serial.print ("Button:");
                     
                      if(button==1)
                      {
                          Serial.println ("not pressed");
                      }
                      else
                      {
                          Serial.println ("pressed");
                      }
                      delay (200);
                    }



Code aansturen met bluetooth module



De code voor het aansturen via bluetooth werkt heel anders. Als er een verbinding is, komen er via de app verschillende letters binnen voor elke knop. In dit programma wordt er uitgelezen of die letter overeenkomt met de geprogrammeerde letters, en wordt er telkens een andere actie gedaan afhankelijk van de letter. 


          

                char dataIn;
                
                int motor1pin1 = 2;
                int motor1pin2 = 3;
                
                int motor2pin1 = 4;
                int motor2pin2 = 5;
                
                
                void setup(){
                  Serial.begin(9600);
                  pinMode(motor1pin1, OUTPUT);
                  pinMode(motor1pin2, OUTPUT);
                  pinMode(motor2pin1, OUTPUT);
                  pinMode(motor2pin2, OUTPUT);
                  Serial.println("start");
                }
                
                void loop(){
                  while(Serial.available() > 0){
                    dataIn = Serial.read();
                    if (dataIn == 'v'){
                      digitalWrite(motor1pin1, HIGH);
                      digitalWrite(motor1pin2, LOW);
                      digitalWrite(motor2pin1, LOW);
                      digitalWrite(motor2pin2, LOW);
                      Serial.println("rijden");
                       }
                    if (dataIn == 'a'){
                      digitalWrite(motor1pin1, LOW);
                      digitalWrite(motor1pin2, HIGH);
                      digitalWrite(motor2pin1, LOW);
                      digitalWrite(motor2pin2, LOW);
                      Serial.println("naar achter");
                    }
                    if (dataIn == 'r'){
                      digitalWrite(motor1pin1, LOW);
                      digitalWrite(motor1pin2, LOW);
                      digitalWrite(motor2pin1, LOW);
                      digitalWrite(motor2pin2, HIGH);
                      Serial.println("naar rechts");
                    }
                    if (dataIn == 'l'){
                      digitalWrite(motor1pin1, LOW);
                      digitalWrite(motor1pin2, LOW);
                      digitalWrite(motor2pin1, HIGH);
                      digitalWrite(motor2pin2, LOW);
                      Serial.println("naar links");
                    }
                    }
                    if (dataIn == 'e'){
                      digitalWrite(motor1pin1, HIGH);
                      digitalWrite(motor1pin2, LOW);
                      digitalWrite(motor2pin1, LOW);
                      digitalWrite(motor2pin2, HIGH);      
                    }
                    if (dataIn == 'q'){
                      digitalWrite(motor1pin1, HIGH);
                      digitalWrite(motor1pin2, LOW);
                      digitalWrite(motor2pin1, HIGH);
                      digitalWrite(motor2pin2, LOW);        
                    }
                    if (dataIn == 'z'){
                      digitalWrite(motor1pin1, LOW);
                      digitalWrite(motor1pin2, HIGH);
                      digitalWrite(motor2pin1, HIGH);
                      digitalWrite(motor2pin2, LOW);        
                    }
                    if (dataIn == 'x'){
                      digitalWrite(motor1pin1, LOW);
                      digitalWrite(motor1pin2, HIGH);
                      digitalWrite(motor2pin1, LOW);
                      digitalWrite(motor2pin2, HIGH);      
                    }
                    if (dataIn == 's'){

                      digitalWrite(motor1pin2, LOW);
                      digitalWrite(motor2pin1, LOW);
